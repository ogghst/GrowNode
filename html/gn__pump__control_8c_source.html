51 void gn_pump_control_task_event_handler(void *handler_args,
52  esp_event_base_t base, int32_t event_id, void *event_data) {
53 
54  ESP_LOGD(TAG, "Pump Control received event: %d", event_id);
55 
56  //gn_leaf_config_handle_t leaf_config = (gn_leaf_config_handle_t) handler_args;
57 
58  gn_leaf_parameter_event_t *evt = (gn_leaf_parameter_event_t*) event_data;
59  switch (event_id) {
60  case GN_LEAF_PARAM_CHANGED_EVENT:
61 
62  //event is from temp sensor
63  if (strcmp(evt->leaf_name, "ds18b20") == 0) {
64 
65  //search for temp parameter
66  for (int i = 0; i < GN_DS18B20_MAX_SENSORS; i++) {
67 
68  if (evt->data != NULL
69  && strcmp(evt->param_name,
70  GN_DS18B20_PARAM_SENSOR_NAMES[i]) == 0) {
71 
72  //double *temp = (double*) evt->data;
73  double temp = strtod(evt->data, NULL);
74 
75  ESP_LOGD(TAG,
76  "parameter changed on leaf: %s - param %s - value %f",
77  evt->leaf_name, GN_DS18B20_PARAM_SENSOR_NAMES[i],
78  temp);
79 
80  //if temp is > 25 and pump is off, turn on
81  if (pump_status == 0 && temp > 26) {
82  //send message to pump
83  if (gn_send_leaf_param_change_message("pump",
84  GN_PUMP_PARAM_TOGGLE, (const char*) &"1", 2)
85  != GN_RET_OK) {
86  gn_log(TAG, GN_LOG_ERROR,
87  "impossible to update parameter %s on leaf %s",
88  GN_PUMP_PARAM_TOGGLE, "pump");
89  return;
90  }
91 
92  ESP_LOGD(TAG, "setting pump to 1");
93 
94  pump_status = 1;
95  }
96 
97  //if temp is < 25 and pump is on, turn off
98  if (pump_status == 1 && temp < 26) {
99  //send message to pump
100  if (gn_send_leaf_param_change_message("pump",
101  GN_PUMP_PARAM_TOGGLE, &"0", 2) != GN_RET_OK) {
102  gn_log(TAG, GN_LOG_ERROR,
103  "impossible to update parameter %s on leaf %s",
104  GN_PUMP_PARAM_TOGGLE, "pump");
105  return;
106  }
107 
108  ESP_LOGD(TAG, "setting pump to 0");
109 
110  pump_status = 0;
111  }
112  }
113  }
114  }
115  break;
116  }

118 }
