95 void gn_cwl_sensor_collect(gn_leaf_handle_t leaf_config) {
96 
97  char leaf_name[GN_LEAF_NAME_SIZE];
98  gn_leaf_get_name(leaf_config, leaf_name);
99 
100  ESP_LOGD(TAG, "[%s] gn_cwl_sensor_collect", leaf_name);
101 
102  //retrieves status descriptor from config
103  //gn_cwl_data_t *data =
104  // (gn_cwl_data_t*) gn_leaf_get_descriptor(leaf_config)->data;
105 
106  double channel;
107  gn_leaf_param_get_double(leaf_config, GN_CWL_PARAM_TOUCH_CHANNEL, &channel);
108 
109  uint16_t read_value;
110  touch_pad_clear_status();
111  touch_pad_read_raw_data(channel, &read_value);
112  //converting to double
113  double result = read_value;
114  //memcpy(&result, &read_value, sizeof(uint16_t));
115 
116  ESP_LOGD(TAG, "[%s] result: %f", leaf_name, result);
117 
118  //store parameter and notify network
119  gn_leaf_param_write_double(leaf_config, GN_CWL_PARAM_ACT_LEVEL, result);
120 
121  double max_level;
122  gn_leaf_param_get_double(leaf_config, GN_CWL_PARAM_MAX_LEVEL, &max_level);
123 
124  double min_level;
125  gn_leaf_param_get_double(leaf_config, GN_CWL_PARAM_MIN_LEVEL, &min_level);
126 
127  bool trg_high;
128  gn_leaf_param_get_bool(leaf_config, GN_CWL_PARAM_TRG_HIGH, &trg_high);
129 
130  bool trg_low;
131  gn_leaf_param_get_bool(leaf_config, GN_CWL_PARAM_TRG_LOW, &trg_low);
132 
133  //if level is above maximum, trigger max
134  if (result >= max_level && trg_high == false) {
135  gn_leaf_param_write_bool(leaf_config, GN_CWL_PARAM_TRG_HIGH,
136  true);
137  } else
138 
139  //reset the trigger status if returning below max
140  if (result < max_level && trg_high == true) {
141  gn_leaf_param_write_bool(leaf_config, GN_CWL_PARAM_TRG_HIGH,
142  false);
143  } else
144 
145  //if level is below maximum, trigger min
146  if (result <= min_level && trg_low == false) {
147  gn_leaf_param_write_bool(leaf_config, GN_CWL_PARAM_TRG_LOW,
148  true);
149  } else
150 
151  //reset the trigger status if returning above min
152  if (result > min_level && trg_low == true) {
153  gn_leaf_param_write_bool(leaf_config, GN_CWL_PARAM_TRG_LOW,
154  false);
155  }
156 
157 }

392 
393  ret = esp_timer_create(&water_sensor_timer_args, &data->sensor_timer);

424  //event arrived for this node

439  if (updtime < 10)
440  updtime = 10;
441  if (updtime > 600)
442  updtime = 600;
443 
444  update_time_sec = updtime;
445 
446  esp_timer_stop(data->sensor_timer);
447  ret = esp_timer_start_periodic(data->sensor_timer,
448  update_time_sec * 1000000);
449 
450  //execute change
451  gn_leaf_param_write_double(leaf_config,
452  GN_CWL_PARAM_UPDATE_TIME_SEC, updtime);
453 
454  } else if (gn_leaf_event_mask_param(&evt, data->active_param)
455  == 0) {
456 
457  bool prev_active = active;
458  int _active = atoi(evt.data);
459 
460  //execute change
461  gn_leaf_param_write_bool(leaf_config, GN_CWL_PARAM_ACTIVE,
462  _active == 0 ? false : true);
463 
464  active = _active;
465 
466  //stop timer if false
467  if (_active == 0 && prev_active == true) {
468  esp_timer_stop(data->sensor_timer);
469  } else if (_active != 0 && prev_active == false) {
470  esp_timer_start_periodic(data->sensor_timer,
471  update_time_sec * 1000000);
472  }
473 
474  }
475 
476  break;
477 
478  default:
479  break;
480 
481  }
482 
483  }
484

495  snprintf(_buf, 20, "%4.2f", act_level);
496  lv_label_set_text(trg_act_value, _buf);
