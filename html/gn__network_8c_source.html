66 void _gn_wifi_event_handler(void *arg, esp_event_base_t event_base,
67  int32_t event_id, void *event_data) {
68 
69 #ifdef CONFIG_GROWNODE_WIFI_ENABLED
70 
71  ESP_LOGD(TAG, "_gn_wifi_event_handler");
72 
73  if (event_base == WIFI_PROV_EVENT) {
74  switch (event_id) {
75  case WIFI_PROV_START:
76  ESP_LOGD(TAG, "WIFI_PROV_START");
77  gn_log(TAG, GN_LOG_INFO, "Provisioning Started");
78  break;
79  case WIFI_PROV_CRED_RECV: {
80  ESP_LOGD(TAG, "WIFI_PROV_CRED_RECV");
81  wifi_sta_config_t *wifi_sta_cfg = (wifi_sta_config_t*) event_data;
82  ESP_LOGD(TAG,
83  "Received Wi-Fi credentials" "\n\tSSID : %s\n\tPassword : %s",
84  (const char* ) wifi_sta_cfg->ssid,
85  (const char* ) wifi_sta_cfg->password);
86  break;
87  }
88  case WIFI_PROV_CRED_FAIL: {
89  ESP_LOGD(TAG, "WIFI_PROV_CRED_FAIL");
90  wifi_prov_sta_fail_reason_t *reason =
91  (wifi_prov_sta_fail_reason_t*) event_data;
92 
93  gn_log(TAG, GN_LOG_ERROR, "Provisioning failed!\n\tReason : %s",
94  ((*reason == WIFI_PROV_STA_AUTH_ERROR) ?
95  "Wi-Fi station authentication failed" :
96  "Wi-Fi access-point not found"));
97  wifi_prov_mgr_reset_provisioning();
98  gn_reboot();
99  break;
100  }
101  case WIFI_PROV_CRED_SUCCESS:
102  ESP_LOGD(TAG, "WIFI_PROV_CRED_SUCCESS");
103  break;
104  case WIFI_PROV_END:
105  ESP_LOGD(TAG, "WIFI_PROV_END");
106  gn_log(TAG, GN_LOG_INFO, "Provisioning OK");
107  break;
108  default:
109  break;
110  }
111  } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_START) {
112  ESP_LOGD(TAG, "WIFI_EVENT_STA_START");
113  esp_wifi_connect();
114  } else if (event_base == IP_EVENT && event_id == IP_EVENT_STA_GOT_IP) {
115  ESP_LOGD(TAG, "IP_EVENT_STA_GOT_IP");
116  s_retry_num = 0;
117  ip_event_got_ip_t *event = (ip_event_got_ip_t*) event_data;
118 
119  char log[42];
120  uint8_t eth_mac[6];
121  const char ssid_prefix[10];
122  strncpy(ssid_prefix, CONFIG_GROWNODE_PROV_SOFTAP_PREFIX, 10);
123  esp_wifi_get_mac(WIFI_IF_STA, eth_mac);
124  char deviceName[17];
125  snprintf(deviceName, 16, "%s%02X%02X%02X", ssid_prefix, eth_mac[3],
126  eth_mac[4], eth_mac[5]);
127 
128  memcpy(_conf->macAddress, eth_mac, 6);
129  strcpy(_conf->deviceName, deviceName);
130 
131  snprintf(log, 41, "%s-%d.%d.%d.%d", deviceName,
132  IP2STR(&event->ip_info.ip));
133  gn_log(TAG, GN_LOG_DEBUG, log);
134 
135  //ESP_LOGI(TAG, "IP : " IPSTR, IP2STR(&event->ip_info.ip));
136 
137  esp_event_post_to(_conf->event_loop, GN_BASE_EVENT,
138  GN_NET_CONNECTED_EVENT, NULL, 0,
139  portMAX_DELAY);
140 
141  /* Signal main application to continue execution */
142  xEventGroupSetBits(_gn_event_group_wifi, GN_WIFI_CONNECTED_EVENT);
143 
144  } else if (event_base == WIFI_EVENT
145  && event_id == WIFI_EVENT_STA_DISCONNECTED) {
146  ESP_LOGD(TAG, "WIFI_EVENT_STA_DISCONNECTED");
147  wifi_event_sta_disconnected_t *disconnected = (wifi_event_sta_disconnected_t *) event_data;
148  ESP_LOGE(TAG,"Wifi disconnected. reason: %d", disconnected->reason);
149  //ESP_LOGI(TAG, "Disconnected. Connecting to the AP again.");
150 
151  esp_event_post_to(_conf->event_loop, GN_BASE_EVENT,
152  GN_NET_DISCONNECTED_EVENT, NULL, 0,
153  portMAX_DELAY);
154 
155  if (_conf->config_init_params->wifi_retries_before_reset_provisioning
156  == -1
157  || s_retry_num
158  < _conf->config_init_params->wifi_retries_before_reset_provisioning) {
159  ESP_LOGI(TAG, "Retry to connect - attempt %i of %i", s_retry_num, _conf->config_init_params->wifi_retries_before_reset_provisioning);
160  s_retry_num++;
161  esp_wifi_connect();
162  } else {
163  xEventGroupSetBits(_gn_event_group_wifi, GN_WIFI_FAIL_EVENT);
164  }
165  //ESP_LOGI(TAG,"connect to the AP fail");
166 
167  }
168 
169 #endif

173 void _gn_wifi_init_sta(void) {
174 
175 #ifdef CONFIG_GROWNODE_WIFI_ENABLED
176 
177  /* Start Wi-Fi in station mode */
178  ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA));
179  ESP_ERROR_CHECK(esp_wifi_start());
180 
181  EventBits_t bits = xEventGroupWaitBits(_gn_event_group_wifi,
182  GN_WIFI_CONNECTED_EVENT | GN_WIFI_FAIL_EVENT,
183  pdFALSE,
184  pdFALSE,
185  portMAX_DELAY);
186 
187  if (bits & GN_WIFI_CONNECTED_EVENT) {
188  gn_log(TAG, GN_LOG_INFO, "Connected to AP");
189  } else if (bits & GN_WIFI_FAIL_EVENT) {
190  gn_log(TAG, GN_LOG_ERROR, "Failed to connect to AP. Resetting Provisioning Status");
191  wifi_prov_mgr_reset_provisioning();
192  gn_reboot();
193  } else {
194  gn_log(TAG, GN_LOG_ERROR, "UNEXPECTED EVENT");
195  }
196 
197 #endif

241 esp_err_t _gn_init_wifi(gn_config_handle_intl_t conf) {
242 
243 #ifdef CONFIG_GROWNODE_WIFI_ENABLED
244 
245  _conf = conf;
246 
247  esp_err_t ret = ESP_OK;
248 
249  ESP_GOTO_ON_ERROR(esp_netif_init(), fail, TAG, "");
250 
251  _gn_event_group_wifi = xEventGroupCreate();
252 
253  ESP_GOTO_ON_ERROR(
254  esp_event_handler_register(WIFI_PROV_EVENT, ESP_EVENT_ANY_ID, &_gn_wifi_event_handler, NULL),
255  fail, TAG, "");
256 
257  ESP_GOTO_ON_ERROR(
258  esp_event_handler_register(WIFI_EVENT, ESP_EVENT_ANY_ID, &_gn_wifi_event_handler, NULL),
259  fail, TAG, "");
260 
261  ESP_GOTO_ON_ERROR(
262  esp_event_handler_register(IP_EVENT, IP_EVENT_STA_GOT_IP, &_gn_wifi_event_handler, NULL),
263  fail, TAG, "");
264 
265  esp_netif_create_default_wifi_sta();
266 
267 #ifdef CONFIG_GROWNODE_PROV_TRANSPORT_SOFTAP
268  esp_netif_create_default_wifi_ap();
269 #endif /* CONFIG_GROWNODE_PROV_TRANSPORT_SOFTAP */
270 
271  wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
272 
273  ESP_GOTO_ON_ERROR(esp_wifi_init(&cfg), fail, TAG, "");
274 
275  conf->wifi_config = cfg;
276 
277  /* Configuration for the provisioning manager */
278  wifi_prov_mgr_config_t config = {
279  /* What is the Provisioning Scheme that we want ?
280  * wifi_prov_scheme_softap or wifi_prov_scheme_ble */
281 #ifdef CONFIG_GROWNODE_PROV_TRANSPORT_BLE
282  .scheme = wifi_prov_scheme_ble,
283 #endif /* CONFIG_GROWNODE_PROV_TRANSPORT_BLE */
284 #ifdef CONFIG_GROWNODE_PROV_TRANSPORT_SOFTAP
285  .scheme = wifi_prov_scheme_softap,
286 #endif /* CONFIG_GROWNODE_PROV_TRANSPORT_SOFTAP */
287 
288  /* Any default scheme specific event handler that you would
289  * like to choose. Since our example application requires
290  * neither BT nor BLE, we can choose to release the associated
291  * memory once provisioning is complete, or not needed
292  * (in case when device is already provisioned). Choosing
293  * appropriate scheme specific event handler allows the manager
294  * to take care of this automatically. This can be set to
295  * WIFI_PROV_EVENT_HANDLER_NONE when using wifi_prov_scheme_softap*/
296 #ifdef CONFIG_GROWNODE_PROV_TRANSPORT_BLE
297  .scheme_event_handler = WIFI_PROV_SCHEME_BLE_EVENT_HANDLER_FREE_BTDM
298 #endif /* CONFIG_GROWNODE_PROV_TRANSPORT_BLE */
299 #ifdef CONFIG_GROWNODE_PROV_TRANSPORT_SOFTAP
300  .scheme_event_handler = WIFI_PROV_EVENT_HANDLER_NONE
301 #endif /* CONFIG_GROWNODE_PROV_TRANSPORT_SOFTAP */
302  };
303 
304  ESP_GOTO_ON_ERROR(wifi_prov_mgr_init(config), fail, TAG, "");
305 
306  conf->prov_config = config;
307 
308  bool provisioned = false;
309 
310 #ifdef CONFIG_GROWNODE_RESET_PROVISIONED
311  wifi_prov_mgr_reset_provisioning();
312 #endif
313 
314  /* Let's find out if the device is provisioned */
315  ESP_GOTO_ON_ERROR(wifi_prov_mgr_is_provisioned(&provisioned), fail, TAG,
316  "");
317 
318  /* If device is not yet provisioned start provisioning service */
319  if (!provisioned) {
320  ESP_LOGI(TAG, "Starting provisioning");
321 
322  /* What is the Device Service Name that we want
323  * This translates to :
324  * - Wi-Fi SSID when scheme is wifi_prov_scheme_softap
325  * - device name when scheme is wifi_prov_scheme_ble
326  */
327  char service_name[18];
328  _gn_wifi_get_device_service_name(service_name, sizeof(service_name));
329 
330  ESP_LOGD(TAG, "service_name: %s", service_name);
331 
332  /* What is the security level that we want (0 or 1):
333  * - WIFI_PROV_SECURITY_0 is simply plain text communication.
334  * - WIFI_PROV_SECURITY_1 is secure communication which consists of secure handshake
335  * using X25519 key exchange and proof of possession (pop) and AES-CTR
336  * for encryption/decryption of messages.
337  */
338 
339 
340 
341  wifi_prov_security_t security = WIFI_PROV_SECURITY_0;
342 
343  if (_conf->config_init_params->provisioning_security) {
344  security = WIFI_PROV_SECURITY_1;
345  }
346 
347  /* Do we want a proof-of-possession (ignored if Security 0 is selected):
348  * - this should be a string with length > 0
349  * - NULL if not used
350  */
351  const char *pop = _conf->config_init_params->provisioning_password;
352 
353  /* What is the service key (could be NULL)
354  * This translates to :
355  * - Wi-Fi password when scheme is wifi_prov_scheme_softap
356  * - simply ignored when scheme is wifi_prov_scheme_ble
357  */
358  const char *service_key = NULL;
359 
360 #ifdef CONFIG_GROWNODE_PROV_TRANSPORT_BLE
361  /* This step is only useful when scheme is wifi_prov_scheme_ble. This will
362  * set a custom 128 bit UUID which will be included in the BLE advertisement
363  * and will correspond to the primary GATT service that provides provisioning
364  * endpoints as GATT characteristics. Each GATT characteristic will be
365  * formed using the primary service UUID as base, with different auto assigned
366  * 12th and 13th bytes (assume counting starts from 0th byte). The client side
367  * applications must identify the endpoints by reading the User Characteristic
368  * Description descriptor (0x2901) for each characteristic, which contains the
369  * endpoint name of the characteristic */
370  uint8_t custom_service_uuid[] = {
371  /* LSB <---------------------------------------
372  * ---------------------------------------> MSB */
373  0xb4, 0xdf, 0x5a, 0x1c, 0x3f, 0x6b, 0xf4, 0xbf,
374  0xea, 0x4a, 0x82, 0x03, 0x04, 0x90, 0x1a, 0x02,
375  };
376  wifi_prov_scheme_ble_set_service_uuid(custom_service_uuid);
377 #endif /* CONFIG_GROWNODE_PROV_TRANSPORT_BLE */
378 
379  /* An optional endpoint that applications can create if they expect to
380  * get some additional custom data during provisioning workflow.
381  * The endpoint name can be anything of your choice.
382  * This call must be made before starting the provisioning.
383  */
384  wifi_prov_mgr_endpoint_create("custom-data");
385 
386  /* Start provisioning service */
387 
388  ESP_GOTO_ON_ERROR(
389  wifi_prov_mgr_start_provisioning(security, pop, service_name,
390  service_key), fail, TAG, "");
391 
392  /* The handler for the optional endpoint created above.
393  * This call must be made after starting the provisioning, and only if the endpoint
394  * has already been created above.
395  */
396  wifi_prov_mgr_endpoint_register("custom-data",
397  _gn_wifi_custom_prov_data_handler, NULL);
398 
399  /* Uncomment the following to wait for the provisioning to finish and then release
400  * the resources of the manager. Since in this case de-initialization is triggered
401  * by the default event loop handler, we don't need to call the following */
402  wifi_prov_mgr_wait();
403  wifi_prov_mgr_deinit();
404 
405  } else {
406  ESP_LOGI(TAG, "Already provisioned, starting Wi-Fi STA");
407 
408  /* We don't need the manager as device is already provisioned,
409  * so let's release it's resources */
410  wifi_prov_mgr_deinit();
411 
412  /* Start Wi-Fi station */
413  _gn_wifi_init_sta();
414  }
415 
416  ESP_LOGD(TAG, "Wait for Wi-Fi connection");
417 
418  /* Wait for Wi-Fi connection */
419  //xEventGroupWaitBits(_gn_event_group_wifi, GN_WIFI_CONNECTED_EVENT, false,
420  //true, portMAX_DELAY);
421  fail: return ret;
422 
423 #else
424  return ESP_OK;
425 #endif /* CONFIG_GROWNODE_WIFI_ENABLED */
426 
427 }
