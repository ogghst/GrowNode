113 void gn_leaf_status_led_task(gn_leaf_handle_t leaf_config) {
114 
115  char leaf_name[GN_LEAF_NAME_SIZE];
116  gn_leaf_get_name(leaf_config, leaf_name);
117  ESP_LOGD(TAG, "Initializing leaf status led %s..",leaf_name);
118 
119  gn_leaf_parameter_event_t evt;
120 
121  double gpio;
122  gn_leaf_param_get_double(leaf_config, GN_LEAF_STATUS_LED_PARAM_GPIO, &gpio);
123 
124  ESP_LOGD(TAG, "assigning to gpio %d", (int )gpio);
125 
126 //setup led
127  gpio_set_direction((int) gpio, GPIO_MODE_OUTPUT);
128  gpio_set_level((int) gpio, 1);
129 
130 //register for events
131  ESP_ERROR_CHECK(
132  esp_event_handler_instance_register_with(gn_leaf_get_event_loop(leaf_config), GN_BASE_EVENT, GN_EVENT_ANY_ID, gn_leaf_led_status_event_handler, leaf_config, NULL));
133 
134 //task cycle
135  while (true) {
136 
137  //ESP_LOGD(TAG, "task cycle..");
138 
139  //check for messages and cycle every 100ms
140  if (xQueueReceive(gn_leaf_get_event_queue(leaf_config), &evt,
141  pdMS_TO_TICKS(100)) == pdPASS) {
142 
143  ESP_LOGD(TAG, "%s - received message: %d", leaf_name, evt.id);
144 
145  //event arrived for this node
146  switch (evt.id) {
147 
148  //parameter change
149  case GN_LEAF_PARAM_CHANGE_REQUEST_EVENT:
150  //what to do when network is connected
151  break;
152 
153  case GN_NET_CONNECTED_EVENT:
154  //gn_pump_state = GN_PUMP_STATE_RUNNING;
155  blink(gpio, 100, 3);
156  break;
157 
158  //what to do when network is disconnected
159  case GN_NET_DISCONNECTED_EVENT:
160  blink(gpio, 500, 3);
161  break;
162 
163  //what to do when server is connected
164  case GN_SRV_CONNECTED_EVENT:
165  blink(gpio, 100, 5);
166  break;
167 
168  //what to do when server is disconnected
169  case GN_SRV_DISCONNECTED_EVENT:
170  blink(gpio, 500, 3);
171  break;
172 
173  case GN_NET_MSG_RECV:
174  blink(gpio, 200, 1);
175  break;
176 
177  default:
178  break;
179 
180  }
181 
182  }
183 
184  vTaskDelay(1000 / portTICK_PERIOD_MS);
185  }
