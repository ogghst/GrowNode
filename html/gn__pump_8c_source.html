59 gn_leaf_descriptor_handle_t gn_pump_config(gn_leaf_handle_t leaf_config) {
60 
61  gn_leaf_descriptor_handle_t descriptor =
62  (gn_leaf_descriptor_handle_t) malloc(sizeof(gn_leaf_descriptor_t));
63  strncpy(descriptor->type, GN_LEAF_PUMP_TYPE, GN_LEAF_DESC_TYPE_SIZE);
64  descriptor->callback = gn_pump_task;
65  descriptor->status = GN_LEAF_STATUS_NOT_INITIALIZED;
66  descriptor->data = NULL;
67 
68  gn_pump_data_t *data = malloc(sizeof(gn_pump_data_t));
69 
70  //parameter definition. if found in flash storage, they will be created with found values instead of default
71  data->gn_pump_toggle_param = gn_leaf_param_create(leaf_config,
72  GN_PUMP_PARAM_TOGGLE, GN_VAL_TYPE_BOOLEAN,
73  (gn_val_t ) { .b = false }, GN_LEAF_PARAM_ACCESS_ALL,
74  GN_LEAF_PARAM_STORAGE_PERSISTED, NULL);
75  gn_leaf_param_add_to_leaf(leaf_config, data->gn_pump_toggle_param);
76 
77  data->gn_pump_channel_param = gn_leaf_param_create(leaf_config,
78  GN_PUMP_PARAM_CHANNEL, GN_VAL_TYPE_BOOLEAN,
79  (gn_val_t ) { .b = false }, GN_LEAF_PARAM_ACCESS_ALL,
80  GN_LEAF_PARAM_STORAGE_PERSISTED, NULL);
81  gn_leaf_param_add_to_leaf(leaf_config, data->gn_pump_channel_param);
82 
83  data->gn_pump_gpio_toggle_param = gn_leaf_param_create(leaf_config,
84  GN_PUMP_PARAM_GPIO_TOGGLE, GN_VAL_TYPE_DOUBLE,
85  (gn_val_t ) { .d = 32 }, GN_LEAF_PARAM_ACCESS_NETWORK,
86  GN_LEAF_PARAM_STORAGE_PERSISTED, NULL);

91  GN_LEAF_PARAM_ACCESS_ALL, GN_LEAF_PARAM_STORAGE_PERSISTED, NULL);

106 void gn_pump_task(gn_leaf_handle_t leaf_config) {
107 
108  bool need_update = true;
109 
110  gn_leaf_parameter_event_t evt;
111 
112  esp_err_t ret = ESP_OK;
113 
114  //retrieves status descriptor from config
115  gn_pump_data_t *data =
116  (gn_pump_data_t*) gn_leaf_get_descriptor(leaf_config)->data;
117 
118  double gpio_toggle;
119  gn_leaf_param_get_double(leaf_config, GN_PUMP_PARAM_GPIO_TOGGLE,
120  &gpio_toggle);
121 
122  double gpio_power;
123  gn_leaf_param_get_double(leaf_config, GN_PUMP_PARAM_GPIO_POWER,
124  &gpio_power);
125 
126  double power;
127  gn_leaf_param_get_double(leaf_config, GN_PUMP_PARAM_POWER, &power);
128 
129  bool toggle;
130  gn_leaf_param_get_bool(leaf_config, GN_PUMP_PARAM_TOGGLE, &toggle);
131 
132  bool channel;
133  gn_leaf_param_get_bool(leaf_config, GN_PUMP_PARAM_CHANNEL, &channel);
134 
135  //setup toggle
136  gpio_set_direction(gpio_toggle, GPIO_MODE_OUTPUT);
137  gpio_set_level(gpio_toggle, toggle ? 1 : 0);
138 
139  //setup pwm
140 
141  char leaf_name[GN_LEAF_NAME_SIZE];
142  gn_leaf_get_name(leaf_config, leaf_name);
143 
144  ESP_LOGD(TAG, "%s - setting pwm pin %d channel %d",
145  leaf_name, (int )gpio_toggle,
146  (int )channel);
147 
148  mcpwm_unit_t pwm_unit = channel ? MCPWM_UNIT_1 : MCPWM_UNIT_0;
149 
150  /*
151  mcpwm_io_signals_t pwm_signal =
152  data->gn_pump_channel_param->param_val->v.b ? MCPWM1A : MCPWM0A;
153  */
154 
155  mcpwm_timer_t pwm_timer = channel ? MCPWM_TIMER_1 : MCPWM_TIMER_0;
156 
157  mcpwm_generator_t pwm_generator = channel ? MCPWM_GEN_B : MCPWM_GEN_A;
158 
159  mcpwm_pin_config_t pin_config = { .mcpwm0a_out_num = (
160  channel ? 0 : gpio_power), .mcpwm1a_out_num = (
161  channel ? gpio_power : 0)
162 
163  };
164  mcpwm_set_pin(pwm_unit, &pin_config);
165  mcpwm_config_t pwm_config;
166  pwm_config.frequency = 10000; //TODO make configurable
167  pwm_config.cmpr_a = 0.0;
168  pwm_config.cmpr_b = 0.0;
169  pwm_config.counter_mode = MCPWM_UP_COUNTER;
170  pwm_config.duty_mode = MCPWM_DUTY_MODE_0;
171  mcpwm_init(pwm_unit, pwm_timer, &pwm_config); //Configure PWM0A & PWM0B with above settings
172 
173  ESP_LOGD(TAG, "%s - pwm initialized", leaf_name);
174 
175  //setup screen, if defined in sdkconfig
176 #ifdef CONFIG_GROWNODE_DISPLAY_ENABLED
177  lv_obj_t *label_status = NULL;
178  lv_obj_t *label_power = NULL;
179  lv_obj_t *label_title = NULL;
180 
181  if (pdTRUE == gn_display_leaf_refresh_start()) {
182 
183  //parent container where adding elements
184  lv_obj_t *_cnt = (lv_obj_t*) gn_display_setup_leaf(leaf_config);
185 
186  if (_cnt) {
187 
188  //style from the container
189  lv_style_t *style = _cnt->styles->style;
190 
191  lv_obj_set_layout(_cnt, LV_LAYOUT_GRID);
192  lv_coord_t col_dsc[] = { 90, LV_GRID_FR(1), LV_GRID_TEMPLATE_LAST };
193  lv_coord_t row_dsc[] = { 20, 20, 20, LV_GRID_FR(1),
194  LV_GRID_TEMPLATE_LAST };
195  lv_obj_set_grid_dsc_array(_cnt, col_dsc, row_dsc);
196 
197  label_title = lv_label_create(_cnt);
198  lv_label_set_text(label_title, leaf_name);
199  //lv_obj_add_style(label_title, style, 0);
200  //lv_obj_align_to(label_title, _cnt, LV_ALIGN_TOP_MID, 0, LV_PCT(10));
201  lv_obj_set_grid_cell(label_title, LV_GRID_ALIGN_CENTER, 0, 2,
202  LV_GRID_ALIGN_STRETCH, 0, 1);
203 
204  label_status = lv_label_create(_cnt);
205  //lv_obj_add_style(label_status, style, 0);
206  lv_label_set_text(label_status,
207  toggle ?
208  "status: on" : "status: off");
209  //lv_obj_align_to(label_status, label_title, LV_ALIGN_BOTTOM_LEFT,
210  // LV_PCT(10), LV_PCT(10));
211  lv_obj_set_grid_cell(label_status, LV_GRID_ALIGN_STRETCH, 0, 1,
212  LV_GRID_ALIGN_STRETCH, 1, 2);
213 
214  label_power = lv_label_create(_cnt);
215  lv_obj_add_style(label_power, style, 0);
216 
217  char _p[21];
218  snprintf(_p, 20, "power: %4.0f",
219  power);
220  lv_label_set_text(label_power, _p);
221  //lv_obj_align_to(label_power, label_status, LV_ALIGN_TOP_LEFT, 0,
222  // LV_PCT(10));
223  lv_obj_set_grid_cell(label_power, LV_GRID_ALIGN_STRETCH, 0, 1,
224  LV_GRID_ALIGN_STRETCH, 2, 2);
225 
226  }
227 
228  gn_display_leaf_refresh_end();
229 
230  }
231 
232 #endif
233 
234  //task cycle
235  while (true) {
236 
237  //check for messages and cycle every 100ms
238  if (xQueueReceive(gn_leaf_get_event_queue(leaf_config), &evt,
239  pdMS_TO_TICKS(100)) == pdPASS) {
240 
241  ESP_LOGD(TAG, "received message: %d", evt.id);
242 
243  //event arrived for this node
244  switch (evt.id) {
245 
246  //parameter change
247  case GN_LEAF_PARAM_CHANGE_REQUEST_EVENT:
248 
249  ESP_LOGD(TAG, "request to update param %s, data = '%s'",
250  evt.param_name, evt.data);
251 
252  //parameter is status
253  if (gn_leaf_event_mask_param(&evt,
254  data->gn_pump_toggle_param) == 0) {
255 
256  const bool ret =
257  strncmp((char*) evt.data, "0", evt.data_size) == 0 ?
258  false : true;
259 
260  //execute change
261  gn_leaf_param_write_bool(leaf_config, GN_PUMP_PARAM_TOGGLE,
262  ret);
263 
264  need_update = true;
265 
266 #ifdef CONFIG_GROWNODE_DISPLAY_ENABLED
267  if (pdTRUE == gn_display_leaf_refresh_start()) {
268 
269  lv_label_set_text(label_status,
270  toggle ?
271  "status: on" : "status: off");
272 
273  gn_display_leaf_refresh_end();
274  }
275 #endif
276 
277  //parameter is power
278  } else if (gn_leaf_event_mask_param(&evt,
279  data->gn_pump_gpio_toggle_param) == 0) {
280 
281  //check limits
282  int gpio = atoi(evt.data);
283  if (gpio >= 0 && gpio < GPIO_NUM_MAX) {
284  //execute change. this will have no effects until restart
285  gn_leaf_param_write_double(leaf_config,
286  GN_PUMP_PARAM_GPIO_TOGGLE, gpio);
287  }
288 
289  } else if (gn_leaf_event_mask_param(&evt,
290  data->gn_pump_power_param) == 0) {
291 
292  double pow = strtod(evt.data, NULL);
293  if (pow < 0)
294  pow = 0;
295  if (pow > 100)
296  pow = 100;
297 
298  //execute change
299  gn_leaf_param_write_double(leaf_config, GN_PUMP_PARAM_POWER,
300  pow);
301 
302  need_update = true;
303 
304 #ifdef CONFIG_GROWNODE_DISPLAY_ENABLED
305  if (pdTRUE == gn_display_leaf_refresh_start()) {
306  char _p[21];
307  snprintf(_p, 20, "power: %f", pow);
308  lv_label_set_text(label_power, _p);
309 
310  gn_display_leaf_refresh_end();
311  }
312 #endif
313 
314  } else if (gn_leaf_event_mask_param(&evt,
315  data->gn_pump_gpio_power_param) == 0) {
316 
317  //check limits
318  int gpio = atoi(evt.data);
319  if (gpio >= 0 && gpio < GPIO_NUM_MAX) {
320  //execute change. this will have no effects until restart
321  gn_leaf_param_write_double(leaf_config,
322  GN_PUMP_PARAM_GPIO_POWER, gpio);
323  }
324 
325  }
326 
327  break;
328 
329  //what to do when network is connected
330  case GN_NET_CONNECTED_EVENT:
331  //gn_pump_state = GN_PUMP_STATE_RUNNING;
332  break;
333 
334  //what to do when network is disconnected
335  case GN_NET_DISCONNECTED_EVENT:
336  break;
337 
338  //what to do when server is connected
339  case GN_SRV_CONNECTED_EVENT:
340  break;
341 
342  //what to do when server is disconnected
343  case GN_SRV_DISCONNECTED_EVENT:
344  break;
345 
346  default:
347  break;
348 
349  }
350 
351  }
352 
353  if (need_update == true) {
354 
355  //finally, we update sensor using the parameter values
356  gn_leaf_param_get_bool(leaf_config, GN_PUMP_PARAM_TOGGLE, &toggle);
357  gn_leaf_param_get_double(leaf_config, GN_PUMP_PARAM_POWER, &power);
358 
359  if (!toggle) {
360  ret = gpio_set_level(gpio_toggle, 0);
361 
362  if (ret != ESP_OK) {
363  gn_log(TAG, GN_LOG_ERROR, "error in disabling signal, channel %d",
364  (int )channel);
365  }
366 
367  ret = mcpwm_set_duty(pwm_unit, pwm_timer, pwm_generator, 0);
368 
369  if (ret != ESP_OK) {
370  gn_log(TAG, GN_LOG_ERROR, "error in changing power, channel %d",
371  (int )channel);
372  }
373 
374  ESP_LOGD(TAG, "%s - toggle off",
375  leaf_name);
376 
377  need_update = false;
378 
379  } else {
380 
381  ret = gpio_set_level(gpio_toggle, 1);
382 
383  if (ret != ESP_OK) {
384  gn_log(TAG, GN_LOG_ERROR, "error in setting signal, channel %d",
385  (int )channel);
386  }
387 
388  ret = mcpwm_set_duty(pwm_unit, pwm_timer, pwm_generator, power);
389 
390  if (ret != ESP_OK) {
391  gn_log(TAG, GN_LOG_ERROR, "error in changing power, channel %d",
392  (int )channel);
393  }
394 
395  ESP_LOGD(TAG, "%s - setting power pin %d to %d",
396  leaf_name, (int )gpio_power,
397  (int )power);
398  need_update = false;
399  }
400  }
401 
402  vTaskDelay(1000 / portTICK_PERIOD_MS);
403  }
404 
405 }
