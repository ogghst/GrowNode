86 static void scroll_left_button_event_handler(lv_event_t *e) {
87 
88  ESP_LOGD(TAG, "scroll_left_button_event_handler");
89 
90  if (e->code != LV_EVENT_CLICKED)
91  return;
92 
93  if (!_config || !_config->node_config || !_current_leaf
94  || _config->node_config->leaves.last < 2)
95  return;
96 
97  ESP_LOGD(TAG, "current leaf: %s", _current_leaf->name);
98 
99  size_t found = 0;
100 
101  for (size_t i = 0; i < _config->node_config->leaves.last; i++) {
102 
103  gn_leaf_config_handle_intl_t _p =
104  (gn_leaf_config_handle_intl_t) _config->node_config->leaves.at[i];
105  ESP_LOGD(TAG, "checking %s", _p->name);
106  if (strcmp(_p->name, _current_leaf->name) == 0) {
107  ESP_LOGD(TAG, "found leaf at %d", i);
108  found = 1;
109  lv_obj_add_flag(_current_leaf->display_container,
110  LV_OBJ_FLAG_HIDDEN);
111  //show the previous panel
112  if (i == 0)
113  _current_leaf =
114  _config->node_config->leaves.at[_config->node_config->leaves.last
115  - 1];
116  else if (i > 0)
117  _current_leaf = _config->node_config->leaves.at[i - 1];
118 
119  ESP_LOGD(TAG, "new leaf %s", _current_leaf->name);
120 
121  break;
122  }
123 
124  }
125  //make it visible if found
126  if (found == 1 && _current_leaf->display_container != NULL)
127  lv_obj_clear_flag(_current_leaf->display_container, LV_OBJ_FLAG_HIDDEN);
128 
129 }
