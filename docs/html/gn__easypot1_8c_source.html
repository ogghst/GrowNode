42 void moisture_callback(const gn_leaf_handle_t moist) {
43 
44  double moist_act;
45  gn_leaf_param_get_double(moist, GN_CMS_PARAM_ACT_LEVEL, &moist_act);
46  gn_log(TAG, GN_LOG_INFO, "easypot1 - measuring moisture: %f", moist_act);
47 
48  //turn on the LED if low with a specific frequency
49  if (moist_act < moist_min && moist_last >= moist_min) {
50  gn_leaf_param_set_bool(led_moist, GN_LED_PARAM_TOGGLE, true);
51  gn_leaf_param_set_double(led_moist, GN_LED_PARAM_BLINK_TIME_MS,
52  blink_time_low);
53  }
54 
55  //turn on the LED if high with a specific frequency
56  else if (moist_act > moist_max && moist_last <= moist_max) {
57  gn_leaf_param_set_bool(led_moist, GN_LED_PARAM_TOGGLE, true);
58  gn_leaf_param_set_double(led_moist, GN_LED_PARAM_BLINK_TIME_MS,
59  blink_time_high);
60  }
61 
62  //turn off the LED if under normal threshold
63  else if (moist_act <= moist_max && moist_act >= moist_min) {
64  gn_leaf_param_set_bool(led_moist, GN_LED_PARAM_TOGGLE, false);
65  }
66 
67  moist_last = moist_act;
68 
69 }

104 void gn_configure_easypot1(gn_node_handle_t node) {
105 
106  //leaves
107  //esp_log_level_set("gn_leaf_led", ESP_LOG_INFO);
108 
109  //creates the moisture sensor
110  moist = gn_leaf_create(node, "moist", gn_capacitive_moisture_sensor_config,
111  4096);
112  //set the channel 4
113  gn_leaf_param_init_double(moist, GN_CMS_PARAM_ADC_CHANNEL, 4); //GPIO12
114  //set update time
115  gn_leaf_param_init_double(moist, GN_CMS_PARAM_UPDATE_TIME_SEC, 5);
116  //set initial status to active (on)
117  gn_leaf_param_init_bool(moist, GN_CMS_PARAM_ACTIVE, true);
118 
119  //creates the temperature sensor
120  temp = gn_leaf_create(node, "temp", gn_ds18b20_config, 4096);
121  //set GPIO
122  gn_leaf_param_init_double(temp, GN_DS18B20_PARAM_GPIO, 0);
123  //set update time
124  gn_leaf_param_init_double(temp, GN_DS18B20_PARAM_UPDATE_TIME_SEC, 5);
125  //set initial status to active (on)
126  gn_leaf_param_init_bool(temp, GN_DS18B20_PARAM_ACTIVE, true);
127 
128  //create the temp led leaf
129  led_temp = gn_leaf_create(node, "led_temp", gn_led_config, 4096);
130  gn_leaf_param_init_double(led_temp, GN_LED_PARAM_GPIO, 1);
131 
132  //create the moisture led leaf
133  led_moist = gn_leaf_create(node, "led_moist", gn_led_config, 4096);
134  gn_leaf_param_init_double(led_moist, GN_LED_PARAM_GPIO, 2);
135 
136  //creates a timer that checks moisture every seconds, using esp_timer API
137  esp_timer_handle_t timer_moisture_handler;
138  esp_timer_create_args_t timer_moisture_args = { .callback =
139  &moisture_callback, .name = "moist_timer" };
140  esp_timer_create(&timer_moisture_args, &timer_moisture_handler);
141  esp_timer_start_periodic(timer_moisture_handler, 1 * 1000000);
142 
143  //creates a timer that checks temperature every seconds, using esp_timer API
144  esp_timer_handle_t timer_temp_handler;
145  esp_timer_create_args_t timer_temp_args = { .callback = &temp_callback,
146  .name = "temp_timer" };
147  esp_timer_create(&timer_temp_args, &timer_temp_handler);
148  esp_timer_start_periodic(timer_temp_handler, 1 * 1000000);
149 
150 }
