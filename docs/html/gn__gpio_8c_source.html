134 void gn_gpio_task(gn_leaf_handle_t leaf_config) {
135 
136  char leaf_name[GN_LEAF_NAME_SIZE];
137  gn_leaf_get_name(leaf_config, leaf_name);
138 
139  ESP_LOGD(TAG, "Initializing gpio leaf %s..", leaf_name);
140 
141  const size_t GN_GPIO_STATE_STOP = 0;
142  const size_t GN_GPIO_STATE_RUNNING = 1;
143 
144  size_t gn_gpio_state = GN_GPIO_STATE_RUNNING;
145  gn_leaf_parameter_event_t evt;
146 
147  //retrieves status descriptor from config
148  gn_gpio_data_t *data =
149  (gn_gpio_data_t*) gn_leaf_get_descriptor(leaf_config)->data;
150 
151  double gpio;
152  gn_leaf_param_get_double(leaf_config, GN_GPIO_PARAM_GPIO, &gpio);
153 
154  bool status;
155  gn_leaf_param_get_bool(leaf_config, GN_GPIO_PARAM_TOGGLE, &status);
156 
157  bool inverted;
158  gn_leaf_param_get_bool(leaf_config, GN_GPIO_PARAM_INVERTED, &inverted);
159 
160  ESP_LOGD(TAG, "configuring - gpio %d, status %d, inverted %d", (int )gpio,
161  status ? 1 : 0, inverted ? 1 : 0);
162 
163  //setup
164  gpio_set_direction((int) gpio, GPIO_MODE_OUTPUT);
165  gpio_set_level((int) gpio,
166  status ? (inverted ? 0 : 1) : (inverted ? 1 : 0));
167 
168  //setup screen, if defined in sdkconfig
169 #ifdef CONFIG_GROWNODE_DISPLAY_ENABLED
170  lv_obj_t *label_status = NULL;
171  lv_obj_t *label_title = NULL;
172 
173  if (pdTRUE == gn_display_leaf_refresh_start()) {
174 
175  //parent container where adding elements
176  lv_obj_t *_cnt = (lv_obj_t*) gn_display_setup_leaf(leaf_config);
177 
178  if (_cnt) {
179 
180  //style from the container
181  lv_style_t *style = _cnt->styles->style;
182 
183  lv_obj_set_layout(_cnt, LV_LAYOUT_GRID);
184  lv_coord_t col_dsc[] = { 90, LV_GRID_FR(1), LV_GRID_TEMPLATE_LAST };
185  lv_coord_t row_dsc[] = { 20, 20, 20, LV_GRID_FR(1),
186  LV_GRID_TEMPLATE_LAST };
187  lv_obj_set_grid_dsc_array(_cnt, col_dsc, row_dsc);
188 
189  label_title = lv_label_create(_cnt);
190  lv_label_set_text(label_title, leaf_name);
191  //lv_obj_add_style(label_title, style, 0);
192  //lv_obj_align_to(label_title, _cnt, LV_ALIGN_TOP_MID, 0, LV_PCT(10));
193  lv_obj_set_grid_cell(label_title, LV_GRID_ALIGN_CENTER, 0, 2,
194  LV_GRID_ALIGN_STRETCH, 0, 1);
195 
196  label_status = lv_label_create(_cnt);
197  //lv_obj_add_style(label_status, style, 0);
198  lv_label_set_text(label_status,
199  "status: off");
200  //lv_obj_align_to(label_status, label_title, LV_ALIGN_BOTTOM_LEFT,
201  // LV_PCT(10), LV_PCT(10));
202  lv_obj_set_grid_cell(label_status, LV_GRID_ALIGN_STRETCH, 0, 1,
203  LV_GRID_ALIGN_STRETCH, 1, 2);
204 
205  }
206 
207  gn_display_leaf_refresh_end();
208 
209  }
210 
211 #endif
212 
213  //task cycle
214  while (true) {
215 
216  //ESP_LOGD(TAG, "task cycle..");
217 
218  //check for messages and cycle every 100ms
219  if (xQueueReceive(gn_leaf_get_event_queue(leaf_config), &evt,
220  pdMS_TO_TICKS(100)) == pdPASS) {
221 
222  ESP_LOGD(TAG, "%s - received message: %d", leaf_name, evt.id);
223 
224  //event arrived for this node
225  switch (evt.id) {
226 
227  //parameter change
228  case GN_LEAF_PARAM_CHANGE_REQUEST_EVENT:
229 
230  ESP_LOGD(TAG, "request to update param %s, data = '%s'",
231  evt.param_name, evt.data);
232 
233  //parameter is status
234  if (gn_leaf_event_mask_param(&evt, data->gn_gpio_status_param)
235  == 0) {
236 
237  int _active = atoi(evt.data);
238 
239  //notify change
240  gn_leaf_param_write_bool(leaf_config, GN_GPIO_PARAM_TOGGLE,
241  _active == 0 ? false : true);
242 
243  status = _active;
244 
245  ESP_LOGD(TAG, "%s - gpio %d, toggle %d, inverted %d",
246  leaf_name, (int )gpio, status ? 1 : 0,
247  inverted ? 1 : 0);
248 
249  //update sensor using the parameter values
250  if (gn_gpio_state == GN_GPIO_STATE_RUNNING) {
251  gpio_set_level((int) gpio,
252  status ?
253  (inverted ? 0 : 1) :
254  (inverted ? 1 : 0));
255  }
256 
257 #ifdef CONFIG_GROWNODE_DISPLAY_ENABLED
258  if (pdTRUE == gn_display_leaf_refresh_start()) {
259 
260  lv_label_set_text(label_status,
261  status ?
262  "status: on" : "status: off");
263 
264  gn_display_leaf_refresh_end();
265  }
266 #endif
267 
268  } else if (gn_leaf_event_mask_param(&evt,
269  data->gn_gpio_inverted_param) == 0) {
270 
271  int _inverted = atoi(evt.data);
272 
273  //notify change
274  gn_leaf_param_write_bool(leaf_config, GN_GPIO_PARAM_INVERTED,
275  _inverted == 0 ? false : true);
276 
277  inverted = _inverted;
278 
279  ESP_LOGD(TAG, "%s - gpio %d, toggle %d, inverted %d",
280  leaf_name, (int )gpio, status ? 1 : 0,
281  inverted ? 1 : 0);
282 
283  //update sensor using the parameter values
284  if (gn_gpio_state == GN_GPIO_STATE_RUNNING) {
285  gpio_set_level((int) gpio,
286  status ?
287  (inverted ? 0 : 1) :
288  (inverted ? 1 : 0));
289  }
290 
291 #ifdef CONFIG_GROWNODE_DISPLAY_ENABLED
292  if (pdTRUE == gn_display_leaf_refresh_start()) {
293 
294  lv_label_set_text(label_status,
295  status ?
296  "status: on" : "status: off");
297 
298  gn_display_leaf_refresh_end();
299  }
300 #endif
301 
302  }
303 
304  break;
305 
306  //what to do when network is connected
307  case GN_NET_CONNECTED_EVENT:
308  //gn_pump_state = GN_PUMP_STATE_RUNNING;
309  break;
310 
311  //what to do when network is disconnected
312  case GN_NET_DISCONNECTED_EVENT:
313  gn_gpio_state = GN_GPIO_STATE_STOP;
314  break;
315 
316  //what to do when server is connected
317  case GN_SRV_CONNECTED_EVENT:
318  gn_gpio_state = GN_GPIO_STATE_RUNNING;
319  break;
320 
321  //what to do when server is disconnected
322  case GN_SRV_DISCONNECTED_EVENT:
323  gn_gpio_state = GN_GPIO_STATE_STOP;
324  break;
325 
326  default:
327  break;
328 
329  }
330 
331  }
332 
333  vTaskDelay(1000 / portTICK_PERIOD_MS);
334  }
