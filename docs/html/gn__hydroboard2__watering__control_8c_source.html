115 gn_leaf_param_validator_result_t _gn_hb2_watering_interval_validator(
116  gn_leaf_param_handle_t param, void **param_value) {
117 
118  double val;
119  if (gn_leaf_param_get_value(param, &val) != GN_RET_OK)
120  return GN_LEAF_PARAM_VALIDATOR_ERROR;
121 
122  double _p1 = **(double**) param_value;
123  ESP_LOGD(TAG, "_watering_interval_validator - param: %d", (int )_p1);
124 
125  if (GN_HYDROBOARD2_MIN_WATERING_INTERVAL > **(double**) param_value) {
126  memcpy(param_value, &GN_HYDROBOARD2_MIN_WATERING_INTERVAL,
127  sizeof(GN_HYDROBOARD2_MIN_WATERING_INTERVAL));
128  return GN_LEAF_PARAM_VALIDATOR_BELOW_MIN;
129  } else if (GN_HYDROBOARD2_MAX_WATERING_INTERVAL
130  < **(double**) param_value) {
131  memcpy(param_value, &GN_HYDROBOARD2_MAX_WATERING_INTERVAL,
132  sizeof(GN_HYDROBOARD2_MAX_WATERING_INTERVAL));
133  return GN_LEAF_PARAM_VALIDATOR_ABOVE_MAX;
134  }
135 
136  _p1 = **(double**) param_value;
137  ESP_LOGD(TAG, "_watering_interval_validator - param: %d", (int )_p1);
138 
139  return GN_LEAF_PARAM_VALIDATOR_PASSED;
140 
141 }

229 inline static void _gn_hb2_watering_control_start_hcc_heating(
230  gn_hb2_watering_control_data_t *data) {
231  gn_log(TAG, GN_LOG_INFO, "Start Heating Cycle");
232  gn_leaf_param_set_bool(data->leaf_plt_hot, GN_GPIO_PARAM_TOGGLE, true);
233  gn_leaf_param_set_bool(data->leaf_plt_cool, GN_GPIO_PARAM_TOGGLE, false);
234  gn_leaf_param_set_bool(data->leaf_plt_pump, GN_LEAF_PWM_PARAM_TOGGLE,
235  true);
236  gn_leaf_param_set_bool(data->leaf_plt_fan, GN_LEAF_PWM_PARAM_TOGGLE, true);
237  data->hcc_cycle = HCC_HEATING;
238  //store time of start water temp control cycle
239  struct timeval tv_now;
240  gettimeofday(&tv_now, NULL);
241  data->hcc_cycle_start = (int64_t) tv_now.tv_sec * 1000000L
242  + (int64_t) tv_now.tv_usec;
243 
244 }

263 inline static bool _gn_hb2_watering_control_hcc_temp_low(double p_wat_temp,

283 void _gn_hb2_watering_callback_intl(gn_leaf_handle_t leaf_config) {
284 
285  ESP_LOGD(TAG, "_gn_hb2_watering_callback");
286 
287  //retrieves status descriptor from config
288  gn_leaf_descriptor_handle_t descriptor = gn_leaf_get_descriptor(
289  leaf_config);
290  gn_hb2_watering_control_data_t *data = descriptor->data;
291 
292  if (data->wat_cycle != WAT_WAIT) {
293  ESP_LOGW(TAG,
294  "Another Watering cycle is already active, not possible to start another.");
295  return;
296  }
297 
298  data->wat_cycle = WAT_OFF;
299 
300  gn_err_t ret;
301 
302  bool p_wat_active = false;
303  double p_wat_int_sec = 0;
304  double p_wat_time_sec = 0;
305  double p_wat_t_temp = 0;
306 
307  double p_cwl = 0;
308  bool p_cwl_active = false;
309  bool p_cwl_trg_high = false;
310  bool p_cwl_trg_low = false;
311 
312  double p_wat_temp = 0;
313  double p_plt_temp = 0;
314  bool p_ds18b20_temp_active = false;
315 
316  double p_amb_temp = 0;
317  bool p_amb_temp_active = true; //TODO test, move to false again!
318 
319  bool p_plt_a_status = false;
320  bool p_plt_b_status = false;
321 
322  bool p_plt_pump_toggle = false;
323  double p_plt_pump_power = 0;
324 
325  bool p_wat_pump = false;
326 
327  struct timeval tv_now;
328  int64_t time_us;
329 
330  //gets watering interval, out of the loop to configure it
331  ret = gn_leaf_param_get_double(leaf_config,
332  GN_HYDROBOARD2_WAT_CTR_PARAM_WATERING_INTERVAL_SEC, &p_wat_int_sec);
333  if (ret != GN_RET_OK) {
334  gn_log(TAG, GN_LOG_ERROR, "watering interval not found");
335  }
336 
337  //infinite loop
338  while (true) {
339 
340  vTaskDelay((p_wat_int_sec * 1000L) / portTICK_PERIOD_MS);
341 
342  data->wat_cycle = WAT_OFF;
343 
344  ret = gn_leaf_param_get_bool(leaf_config,
345  GN_HYDROBOARD2_WAT_CTR_PARAM_ACTIVE, &p_wat_active);
346  if (ret != GN_RET_OK) {
347  gn_log(TAG, GN_LOG_ERROR,
348  "watering cycle active parameter not found");
349  }
350 
351  //watering cycle
352  while (p_wat_active) {
353 
354  //check if watering interval is true
355  ret = gn_leaf_param_get_bool(leaf_config,
356  GN_HYDROBOARD2_WAT_CTR_PARAM_ACTIVE, &p_wat_active);
357  if (ret != GN_RET_OK) {
358  gn_log(TAG, GN_LOG_ERROR,
359  "watering cycle active parameter not found");
360  }
361 
362  //gets watering interval
363  ret = gn_leaf_param_get_double(leaf_config,
364  GN_HYDROBOARD2_WAT_CTR_PARAM_WATERING_INTERVAL_SEC,
365  &p_wat_int_sec);
366  if (ret != GN_RET_OK) {
367  gn_log(TAG, GN_LOG_ERROR, "watering interval not found");
368  break;
369  }
370 
371  //gets watering time
372  ret = gn_leaf_param_get_double(leaf_config,
373  GN_HYDROBOARD2_WAT_CTR_PARAM_WATERING_TIME_SEC,
374  &p_wat_time_sec);
375  if (ret != GN_RET_OK) {
376  gn_log(TAG, GN_LOG_ERROR, "watering time not found");
377  break;
378  }
379 
380  //gets watering target temp
381  ret = gn_leaf_param_get_double(leaf_config,
382  GN_HYDROBOARD2_WAT_CTR_PARAM_WATERING_TARGET_TEMP,
383  &p_wat_t_temp);
384  if (ret != GN_RET_OK) {
385  gn_log(TAG, GN_LOG_ERROR, "watering target temp not found");
386  break;
387  }
388 
389  //gets water level sensor active
390  ret = gn_leaf_param_get_bool(leaf_config,
391  GN_HYDROBOARD2_WAT_CTR_PARAM_ACTIVE, &p_cwl_active);
392  if (ret != GN_RET_OK) {
393  gn_log(TAG, GN_LOG_ERROR,
394  "watering level active flag not found");
395  break;
396  }
397 
398  //gets water level sensor and value
399  ret = gn_leaf_param_get_double(data->leaf_cwl,
400  GN_CWL_PARAM_ACT_LEVEL, &p_cwl);
401  if (ret != GN_RET_OK) {
402  gn_log(TAG, GN_LOG_ERROR, "actual water level not found");
403  break;
404  }
405 
406  //gets water too high value
407  ret = gn_leaf_param_get_bool(data->leaf_cwl, GN_CWL_PARAM_TRG_HIGH,
408  &p_cwl_trg_high);
409  if (ret != GN_RET_OK) {
410  gn_log(TAG, GN_LOG_ERROR,
411  "actual water trigger high value not found");
412  break;
413  }
414 
415  //gets water too low value
416  ret = gn_leaf_param_get_bool(data->leaf_cwl, GN_CWL_PARAM_TRG_LOW,
417  &p_cwl_trg_low);
418  if (ret != GN_RET_OK) {
419  gn_log(TAG, GN_LOG_ERROR,
420  "actual water trigger low value not found");
421  break;
422  }
423 
424  //gets db18b20 temperature active
425  ret = gn_leaf_param_get_bool(data->leaf_ds18b20,
426  GN_DS18B20_PARAM_ACTIVE, &p_ds18b20_temp_active);
427  if (ret != GN_RET_OK) {
428  gn_log(TAG, GN_LOG_ERROR,
429  "watering temperature active flag not found");
430  break;
431  }
432 
433  //gets water temperature and value
434  ret = gn_leaf_param_get_double(data->leaf_ds18b20,
435  GN_DS18B20_PARAM_SENSOR_NAMES[1], &p_plt_temp);
436  if (ret != GN_RET_OK) {
437  gn_log(TAG, GN_LOG_ERROR, "water temperature not found");
438  break;
439  }
440 
441  //gets peltier temperature and value
442  ret = gn_leaf_param_get_double(data->leaf_ds18b20,
443  GN_DS18B20_PARAM_SENSOR_NAMES[0], &p_wat_temp);
444  if (ret != GN_RET_OK) {
445  gn_log(TAG, GN_LOG_ERROR, "peltier temperature not found");
446  break;
447  }
448 
449  //gets ambient temperature active flag
450  ret = gn_leaf_param_get_bool(data->leaf_bme280,
451  GN_BME280_PARAM_ACTIVE, &p_amb_temp_active);
452  if (ret != GN_RET_OK) {
453  gn_log(TAG, GN_LOG_ERROR,
454  "ambient temperature active flag not found");
455  break;
456  }
457 
458  //gets ambient temperature and value
459  ret = gn_leaf_param_get_double(data->leaf_bme280,
460  GN_BME280_PARAM_TEMP, &p_amb_temp);
461  if (ret != GN_RET_OK) {
462  gn_log(TAG, GN_LOG_ERROR, "ambient temperature not found");
463  break;
464  }
465 
466  //gets peltier A status
467  ret = gn_leaf_param_get_bool(data->leaf_plt_hot,
468  GN_GPIO_PARAM_TOGGLE, &p_plt_a_status);
469  if (ret != GN_RET_OK) {
470  gn_log(TAG, GN_LOG_ERROR, "peltier A status not found");
471  break;
472  }
473 
474  //gets peltier B status
475  ret = gn_leaf_param_get_bool(data->leaf_plt_cool,
476  GN_GPIO_PARAM_TOGGLE, &p_plt_b_status);
477  if (ret != GN_RET_OK) {
478  gn_log(TAG, GN_LOG_ERROR, "peltier B status not found");
479  break;
480  }
481 
482  //gets water pump status
483  ret = gn_leaf_param_get_bool(data->leaf_wat_pump,
484  GN_LEAF_PWM_PARAM_TOGGLE, &p_wat_pump);
485  if (ret != GN_RET_OK) {
486  gn_log(TAG, GN_LOG_ERROR, "water pump status not found");
487  break;
488  }
489 
490  //gets plt pump status
491  ret = gn_leaf_param_get_bool(data->leaf_plt_pump,
492  GN_LEAF_PWM_PARAM_TOGGLE, &p_plt_pump_toggle);
493  if (ret != GN_RET_OK) {
494  gn_log(TAG, GN_LOG_ERROR, "hcc pump status not found");
495  break;
496  }
497 
498  //gets hcc power status
499  ret = gn_leaf_param_get_double(data->leaf_plt_pump,
500  GN_LEAF_PWM_PARAM_POWER, &p_plt_pump_power);
501  if (ret != GN_RET_OK) {
502  gn_log(TAG, GN_LOG_ERROR, "hcc pump power not found");
503  break;
504  }
505 
506  // watering algorithm start
507  ESP_LOGD(TAG, "Start Watering Check Cycle. Status = %d",
508  (int )data->hcc_cycle);
509 
510  //if some of the sensors are not active, stop every operation and exit
511  if (!p_cwl_active || !p_ds18b20_temp_active) {
512  gn_log(TAG, GN_LOG_WARNING, "Sensors not active");
513  _gn_hb2_watering_control_stop_hcc(data);
514  _gn_hb2_watering_control_stop_watering(data);
515  break;
516 
517  }
518 
519  //check water level. if out of allowable range - send error and end of cycle
520  if (p_cwl_trg_low == true) {
521  gn_log(TAG, GN_LOG_WARNING,
522  "Not Enough Water to start watering cycle");
523  _gn_hb2_watering_control_stop_hcc(data);
524  _gn_hb2_watering_control_stop_watering(data);
525  break;
526 
527  } else if (p_cwl_trg_high == true) {
528  gn_log(TAG, GN_LOG_WARNING,
529  "Water level too high to start watering cycle");
530  _gn_hb2_watering_control_stop_hcc(data);
531  _gn_hb2_watering_control_stop_watering(data);
532  break;
533 
534  }
535 
536  //check need of cooling
537  if (data->hcc_cycle != HCC_COOLING
538  && _gn_hb2_watering_control_hcc_temp_high(p_wat_temp,
539  p_wat_t_temp)) {
540  _gn_hb2_watering_control_start_hcc_cooling(data);
541  }
542 
543  //check need of heating
544  if (data->hcc_cycle != HCC_HEATING
545  && _gn_hb2_watering_control_hcc_temp_low(p_wat_temp,
546  p_wat_t_temp)) {
547  _gn_hb2_watering_control_start_hcc_heating(data);
548  }
549 
550  // make sure to return to normal parameter if temp within target
551  if (data->hcc_cycle != HCC_OFF
552  && _gn_hb2_watering_control_hcc_temp_ok(p_wat_temp,
553  p_wat_t_temp)) {
554  _gn_hb2_watering_control_stop_hcc(data);
555  }
556 
557  //if water temp into normal threshold then start watering
558  if (_gn_hb2_watering_control_hcc_temp_ok(p_wat_temp, p_wat_t_temp)
559  && data->wat_cycle == WAT_OFF) {
560  _gn_hb2_watering_control_start_watering(data);
561  }
562 
563  //check if water temp climate time is below maximum time
564  if (data->hcc_cycle != HCC_OFF) {
565 
566  //get current time
567  gettimeofday(&tv_now, NULL);
568  time_us = (int64_t) tv_now.tv_sec * 1000000L
569  + (int64_t) tv_now.tv_usec;
570 
571  //check last hcc activation
572  if ((time_us - data->hcc_cycle_start)
573  > (((int64_t) GN_HYDROBOARD2_MAX_HCC_CYCLE_TIME_SEC)
574  * 1000000L)) {

577  GN_HYDROBOARD2_MAX_HCC_CYCLE_TIME_SEC);
578  _gn_hb2_watering_control_stop_hcc(data);
579  break;
580  }
581 
582  }
583

585  if (data->wat_cycle == WAT_ON) {

589  data->wat_cycle_cumulative_time_ms);
590  }
591 
592  //if out of maximum watering time - end of cycle

594  gn_log(TAG, GN_LOG_INFO,
595  "Maximum Watering time (%l) reached, ending",
596  p_wat_time_sec);
597  _gn_hb2_watering_control_stop_watering(data);

599  }
600

602  vTaskDelay(
603  GN_HYDROBOARD2_WAT_CTR_CYCLE_TIME_MS / portTICK_PERIOD_MS);
604 
605  }
606

608  //gn_log(TAG, GN_LOG_INFO, "Ending Watering Cycle");
609
